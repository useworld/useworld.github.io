Node.js에서 C 라이브러리 사용하기 - FFI(Foreign Function Interface)

2016년 04월 20일

NodeJS에서 C언어로 만들어진 라이브러리 파일을 이용하는 Test를 해보았다. fft는 fast Fourier transform C 라이브러리다.

var ref = require('ref');
var ffi = require('ffi');
var ArrayType = require('ref-array');

var double = ref.types.double;
var DoubleArray = ArrayType(double);
var fftlib = ffi.Library('./fft', {
    "fft" : [ 'void', [DoubleArray, DoubleArray, 'int', 'int']]
});

var frData = [3.6, 2.9, 5.6, 4.8, 3.3, 5.9, 5, 4.3];   
var fiData = [0, 0, 0, 0, 0, 0, 0, 0];
var n = 8;
var inv = 0;
var fr = new DoubleArray(frData);
var fi = new DoubleArray(fiData);
fftlib.fft(fr, fi, n, inv);

console.log("======== NODEJS =========");
for(var i = 0; i < n; i++) {
    console.log(i+" : " + fr[i]);
}
